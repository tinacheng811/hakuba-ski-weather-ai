from datetime import datetime

def predict_at_date(target_date_str, model, scaler, df, window_size=7):
    target_date = pd.to_datetime(target_date_str)
    last_date = df['Date'].max()


# 1. 
    if target_date <= last_date:
        row = df[df['Date'] == target_date]
        if not row.empty:
            return row['tavg'].values[0], row['tmax'].values[0], row['tmin'].values[0], row['snowf'].values[0]

    # 2. 
    #  9 
    features_cols = ['tavg', 'tmax', 'tmin', 'prcp', 'snowf', 'snowdmax', 'sunhour', 'month_sin', 'month_cos']

    #  numpy
    last_data_raw = df[features_cols].tail(window_size).fillna(0).values

    #  scaler  9 
    last_data_scaled = scaler.transform(last_data_raw)

    current_batch = last_data_scaled.reshape(1, window_size, 9)
    final_pred = None

    days_to_predict = (target_date - last_date).days
    for i in range(days_to_predict):
        #  5 [tavg, tmax, tmin, snowf, snowdmax]
        pred = model.predict(current_batch, verbose=0)[0]

        current_date = last_date + pd.Timedelta(days=i+1)
        m_sin = np.sin(2 * np.pi * current_date.month / 12)
        m_cos = np.cos(2 * np.pi * current_date.month / 12)

        #  9 
        # : [tavg, tmax, tmin, prcp(0), snowf, snowdmax, sunhour(0.5), m_sin, m_cos]
        new_entry = np.array([pred[0], pred[1], pred[2], 0, pred[3], pred[4], 0.5, m_sin, m_cos]).reshape(1, 1, 9)
        current_batch = np.append(current_batch[:, 1:, :], new_entry, axis=1)
        final_pred = pred

    # 3.  ( 9  dummy )
    dummy = np.zeros((1, 9))
    dummy[0, 0:3] = final_pred[0:3] # tavg, tmax, tmin
    dummy[0, 4:6] = final_pred[3:5] # snowf, snowdmax
    res = scaler.inverse_transform(dummy)[0]

    return res[0], res[1], res[2], res[4]

import pandas as pd
import numpy as np

def get_ski_recommendation(start_date_str, end_date_str, model, scaler, df, window_size=7):
    # 1. 
    start_date = pd.to_datetime(start_date_str)
    end_date = pd.to_datetime(end_date_str)
    last_date = df['Date'].max()

    # 2.  ( 9 )
    features_cols = ['tavg', 'tmax', 'tmin', 'prcp', 'snowf', 'snowdmax', 'sunhour', 'month_sin', 'month_cos']

    # 3. 
    predictions_list = []

    # 4. 
    #  df 
    if end_date <= last_date:
        relevant_df = df[(df['Date'] >= start_date) & (df['Date'] <= end_date)]
        for _, row in relevant_df.iterrows():
            predictions_list.append({
                'date': row['Date'],
                'tavg': row['tavg'], 'tmax': row['tmax'], 'tmin': row['tmin'],
                'snowf': row['snowf'], 'snowdmax': row['snowf'] # 
            })
    else:
        #  LSTM 
        days_to_predict = (end_date - last_date).days

        #  window_size 
        last_data_raw = df[features_cols].tail(window_size).fillna(0).values
        current_batch = scaler.transform(last_data_raw).reshape(1, window_size, 9)

        for i in range(days_to_predict):
            #  5 : [tavg, tmax, tmin, snowf, snowdmax]
            pred = model.predict(current_batch, verbose=0)[0]
            curr_date = last_date + pd.Timedelta(days=i+1)

            # 
            if start_date <= curr_date <= end_date:
                dummy = np.zeros((1, 9))
                dummy[0, 0:3] = pred[0:3] # tavg, tmax, tmin
                dummy[0, 4:6] = pred[3:5] # snowf, snowdmax
                res = scaler.inverse_transform(dummy)[0]

                predictions_list.append({
                    'date': curr_date,
                    'tavg': res[0], 'tmax': res[1], 'tmin': res[2],
                    'snowf': res[4], 'snowdmax': res[5]
                })

            #  ( 9 )
            m_sin = np.sin(2 * np.pi * curr_date.month / 12)
            m_cos = np.cos(2 * np.pi * curr_date.month / 12)
            #  [tavg, tmax, tmin, prcp(0), snowf, snowdmax, sunhour(0.5), m_sin, m_cos]
            new_entry = np.array([pred[0], pred[1], pred[2], 0, pred[3], pred[4], 0.5, m_sin, m_cos]).reshape(1, 1, 9)
            current_batch = np.append(current_batch[:, 1:, :], new_entry, axis=1)

    # 5.  ()
# 5. 
    final_scores = []
    for day in predictions_list:
        score = 0
        score += day['snowdmax'] * 1.0  # 
        if day['snowf'] > 2 and day['tmax'] < 0:
            score += 30 # 
        if day['tmax'] > 3:
            score -= 20 # 

        #  (Indentation) 
        star_count = 1
        if score >= 80: star_count = 5
        elif score >= 50: star_count = 4
        elif score >= 20: star_count = 3
        elif score > 0: star_count = 2

        # ---  ---
        final_scores.append({
            'date': day['date'],
            'score': score,
            'info': day,
            'stars': "" * star_count + "" * (5 - star_count),
            'tips': get_travel_tips(star_count, day) # 
        })
    # ------------------

    if not final_scores:
        return None, []

    best_day = max(final_scores, key=lambda x: x['score'])
    return best_day, final_scores
